        folder = filedialog.askdirectory(title="Выберите папку с .xlsb файлами")
        if folder:
            self.source_files = list(Path(folder).glob("*.xlsb"))
            self.update_status(f"Найдено {len(self.source_files)} файлов (.xlsb)")
            self.update_run_button_state()

    def select_files(self):
        files = filedialog.askopenfilenames(
            title="Выберите .xlsb файлы",
            filetypes=[("Excel Binary Files", "*.xlsb")]
        )
        if files:
            self.source_files = [Path(f) for f in files]
            self.update_status(f"Выбрано {len(self.source_files)} файлов")
            self.update_run_button_state()

    def run_merge(self):
        if not self.source_files:
            messagebox.showwarning("Внимание", "Не выбраны исходные файлы (.xlsb).")
            return
        if not self.target_file:
            messagebox.showwarning("Внимание", "Не выбран целевой файл (.xlsx).")
            return

        try:
            # --- Формируем имя результата: ДД.ММ.ГГГГ Имя.xlsx ---
            today_str = datetime.now().strftime("%d.%m.%Y")
            base_name = self.target_file.stem
            new_name = f"{today_str} {base_name}.xlsx"
            output_path = self.target_file.parent / new_name

            # --- Создаём копию целевого файла ---
            import shutil
            shutil.copy2(self.target_file, output_path)
            self.update_status("Копирование целевого файла...", color="orange")

            # --- Открываем копию для редактирования ---
            wb = load_workbook(output_path)
            sheet_name = "ЕЖЕДНЕВНЫЙ УЧЕТ"
            if sheet_name not in wb.sheetnames:
                raise ValueError(f"Лист '{sheet_name}' не найден в файле.")
            ws = wb[sheet_name]

            # Определяем последнюю строку по столбцу G (индекс 7)
            last_row = ws.max_row
            col_g = 7  # G = 7-й столбец (A=1, B=2, ..., G=7)
            while last_row >= 6:
                cell = ws.cell(row=last_row, column=col_g)
                if cell.value and str(cell.value).strip():
                    break
                last_row -= 1
            next_row = last_row + 1

            total_added = 0
            for file_path in self.source_files:
                try:
                    df = pd.read_excel(
                        file_path,
                        sheet_name=sheet_name,
                        header=4,           # строка 5 — заголовки
                        usecols="A:P",      # A–P
                        engine='pyxlsb'
                    )
                except Exception as e:
                    messagebox.showwarning(
                        "Пропущен файл",
                        f"Не удалось прочитать:\n{file_path.name}\nОшибка: {e}"
                    )
                    continue

                # Фильтруем по столбцу 'Гражданство' (если есть)
                if 'Гражданство' in df.columns:
                    mask = df['Гражданство'].notna() & df['Гражданство'].astype(str).str.strip().ne("")
                    df = df[mask]

                # Вставляем строки
                for _, row in df.iterrows():
                    for col_idx, val in enumerate(row, start=1):  # A=1, B=2, ..., P=16
                        ws.cell(row=next_row, column=col_idx, value=val)
                    next_row += 1
                    total_added += 1

            wb.save(output_path)
            wb.close()

            self.update_status(
                f"✅ Готово!\nДобавлено: {total_added} строк\nФайл сохранён:\n{output_path.name}",
                color="green"
            )
            messagebox.showinfo(
                "Успех",
                f"Объединение завершено!\n\nСохранено как:\n{output_path}"
            )

        except Exception as e:
            error_msg = str(e)
            messagebox.showerror("Ошибка", f"Произошла ошибка:\n{error_msg}")
            self.update_status("❌ Ошибка — см. сообщение", color="red")
            import traceback
            print("\n--- ТРЕЙСБЭК ---")
            traceback.print_exc()

class WorkerThread(QThread):
    log_signal = pyqtSignal(str)
    finished_signal = pyqtSignal(int, int)
    error_signal = pyqtSignal(str)
    
    def __init__(self, app_instance, mode, action, selected_dates=None):
        super().__init__()
        self.app = app_instance
        self.mode = mode  # 'all' или 'dates'
        self.action = action  # 'copy' или 'move'
        self.selected_dates = selected_dates or []
    
    def run(self):
        try:
            if self.mode == 'all':
                self.process_all_files()
            else:
                self.process_selected_dates()
        except Exception as e:
            self.error_signal.emit(str(e))
    
    def log(self, message):
        self.log_signal.emit(message)
    
    def process_all_files(self):
        base_path = self.app.folder_path.text()
        rename_folder = os.path.join(base_path, "Rename")
        
        os.makedirs(rename_folder, exist_ok=True)
        self.log(f"Создана папка: {rename_folder}")
        self.log(f"Режим: Обработка всех файлов")
        self.log(f"Действие: {'перемещение' if self.action == 'move' else 'копирование'}")
        
        total_moved = 0
        total_processed = 0
        
        date_pattern = r'\b(\d{2}\.\d{2}\.\d{4})\b'
        
        for file in os.listdir(base_path):
            if file.lower().endswith(('.xlsx', '.xlsb', '.xls', '.xlsm', '.xlt', '.xltx', '.xltm', '.csv')):
                src = os.path.join(base_path, file)
                
                matches = re.findall(date_pattern, file)
                if matches:
                    target_date = matches[0]
                else:
                    target_date = datetime.now().strftime("%d.%m.%Y")
                    self.log(f"В файле {file} не найдена дата, используется текущая: {target_date}")
                
                new_filename = self.app.generate_new_filename(file, target_date)
                temp_dest = os.path.join(rename_folder, file)
                final_dest = os.path.join(rename_folder, new_filename)
                
                try:
                    if self.action == "move":
                        shutil.move(src, temp_dest)
                        self.log(f"Перемещен: {file}")
                    else:
                        shutil.copy2(src, temp_dest)
                        self.log(f"Скопирован: {file}")
                    
                    total_moved += 1
                    
                    self.log(f"Обрабатываю файл: {file}")
                    processed_file = os.path.splitext(temp_dest)[0] + '.xlsx'
                    
                    if self.app.process_excel_file(temp_dest, target_date):
                        total_processed += 1
                        
                        if os.path.exists(processed_file):
                            os.rename(processed_file, final_dest)
                            self.log(f"Переименован в: {new_filename}")
                        
                        if temp_dest != processed_file and os.path.exists(temp_dest):
                            os.remove(temp_dest)
                            
                        self.log(f"Файл успешно обработан: {new_filename}")
                    else:
                        self.log(f"Ошибка обработки файла: {file}")
                        
                except Exception as e:
                    self.log(f"ОШИБКА при обработке {file}: {e}")
        
        self.log(f"Обработка завершена. Всего {'перемещено' if self.action == 'move' else 'скопировано'}: {total_moved} файлов")
        self.log(f"Успешно обработано: {total_processed} файлов")
        self.finished_signal.emit(total_moved, total_processed)
    
    def process_selected_dates(self):
        self.log(f"Начинаю обработку {len(self.selected_dates)} дат: {', '.join(self.selected_dates)}")
        self.log(f"Действие: {'перемещение' if self.action == 'move' else 'копирование'}")
        
        total_moved = 0
        total_processed = 0
        
        for date in self.selected_dates:
            moved, processed = self.organize_files(date)
            total_moved += moved
            total_processed += processed
        
        self.log(f"Обработка завершена. Всего {'перемещено' if self.action == 'move' else 'скопировано'}: {total_moved} файлов")
        self.log(f"Успешно обработано: {total_processed} файлов")
        self.finished_signal.emit(total_moved, total_processed)

        self.log(f"Создана папка: {date_folder}")
        
        moved_count = 0

        for file in os.listdir(base_path):
            if file.lower().endswith(('.xlsx', '.xlsb', '.xls', '.xlsm', '.xlt', '.xltx', '.xltm', '.csv')) and date in file:
                src = os.path.join(base_path, file)
                
                new_filename = self.app.generate_new_filename(file, date)
                temp_dest = os.path.join(date_folder, file)
                final_dest = os.path.join(date_folder, new_filename)
                
                try:
                    if self.action == "move":
                        shutil.move(src, temp_dest)
                        self.log(f"Перемещен: {file}")
                    else:
                        shutil.copy2(src, temp_dest)
                        self.log(f"Скопирован: {file}")
                    
                    moved_count += 1
                    
                    self.log(f"Обрабатываю файл: {file}")
                    if self.app.process_excel_file(temp_dest, date):
                        processed_count += 1
                        
                        processed_file = os.path.splitext(temp_dest)[0] + '.xlsx'
                        if os.path.exists(processed_file):
                            os.rename(processed_file, final_dest)
                            self.log(f"Переименован в: {new_filename}")
                        
                        if temp_dest != processed_file and os.path.exists(temp_dest):
                            os.remove(temp_dest)
                            
                        self.log(f"Файл успешно обработан: {new_filename}")
                    else:

                except Exception as e:

        """Генерирует новое имя файла: ДД.ММ.ГГГГ - [имя без дат].xlsx"""
        name_part, ext = os.path.splitext(original_filename)
        
        date_pattern = r'\d{2}\.\d{2}\.\d{4}'
        cleaned_name = re.sub(date_pattern, '', name_part)
        cleaned_name = cleaned_name.strip(' ._-')
        cleaned_name = re.sub(r'\s+', ' ', cleaned_name)
        
        new_name = f"{target_date} - {cleaned_name}.xlsx"

            selected_dates = [item.text() for item in self.date_listbox.selectedItems()] if not self.all_files_checkbox.isChecked() else []
            
            self.start_processing(mode, action, selected_dates)
    
    def start_processing(self, mode, action, selected_dates):
        self.process_btn.setEnabled(False)
        self.process_btn.setText("⏳ Обработка...")
        
        self.worker = WorkerThread(self, mode, action, selected_dates)
        self.worker.log_signal.connect(self.log)
        self.worker.finished_signal.connect(self.processing_finished)
        self.worker.error_signal.connect(self.processing_error)
        self.worker.start()
    
    def processing_finished(self, total_moved, total_processed):
        self.process_btn.setEnabled(True)
        self.process_btn.setText("⚙️ Сформировать")
        
        action_text = "перемещено" if self.worker.action == 'move' else 'скопировано'
        QMessageBox.information(
            self, 
            "Готово", 
            f"{action_text.capitalize()} {total_moved} файлов\nОбработано {total_processed} файлов"
        )
        self.statusBar().showMessage(f"Обработка завершена: {total_moved} файлов")
    
    def processing_error(self, error_msg):
        self.process_btn.setEnabled(True)
        self.process_btn.setText("⚙️ Сформировать")
        QMessageBox.critical(self, "Ошибка", f"Произошла ошибка: {error_msg}")
        self.log(f"КРИТИЧЕСКАЯ ОШИБКА: {error_msg}")
    
    def process_excel_file(self, file_path, date):
        try:
            file_ext = os.path.splitext(file_path)[1].lower()
            new_path = os.path.splitext(file_path)[0] + '.xlsx'
            
            if file_ext == '.xlsb' and EXCEL_AVAILABLE:
                excel = None
                wb = None
                temp_path = None
                
                try:
                    excel = win32.Dispatch("Excel.Application")
                    excel.Visible = False
                    excel.DisplayAlerts = False
                    
                    temp_path = tempfile.mktemp(suffix='.xlsx')
                    wb = excel.Workbooks.Open(file_path)
                    
                    for ws in wb.Worksheets:
                        if ws.Name == 'ЕЖЕДНЕВНЫЙ УЧЕТ':
                            ws.Cells(1, 2).Value = date
                            ws.Columns(4).Delete()
                            break
                    
                    wb.SaveAs(temp_path, FileFormat=51)
                    
                except Exception as e:
                    self.log(f"Ошибка конвертации XLSB: {e}")
                    return False
                finally:
                    try:
                        if wb: wb.Close(SaveChanges=False)
                    except: pass
                    try:
                        if excel: excel.Application.Quit()
                    except: pass
                    
                    if temp_path and os.path.exists(temp_path):
                        shutil.copy2(temp_path, new_path)
                        os.remove(temp_path)
                
            else:
                wb = load_workbook(file_path)
                if 'ЕЖЕДНЕВНЫЙ УЧЕТ' in wb.sheetnames:
                    ws = wb['ЕЖЕДНЕВНЫЙ УЧЕТ']
                    ws['B1'] = date
                    ws.delete_cols(4)
                wb.save(new_path)
                wb.close()
            
            if file_path != new_path and os.path.exists(file_path):
                os.remove(file_path)
            
            return True
        except Exception as e:
            QMessageBox.critical(self, "Ошибка", f"Ошибка обработки {os.path.basename(file_path)}: {e}")
            return False
