Структура MRZ для TD3 (паспорт)
Пример (реальный из ваших файлов):

text
PNRUSGAFAROV<<RAMIS<RASIMOVI3<<<<<<<<<<<<<<<<
8020980852RUS0506247M<<<<<<<<5250904020033<74
Первая строка (длина 44 символа):

Позиции 1–2: P (тип документа, всегда P для паспорта)

Позиции 3–5: код эмитента (для РФ – RUS)

Позиции 6–44: фамилия и имя, разделённые <. Фамилия до <<, затем имя, затем возможно отчество (разделители <). В конце заполнители <.

Вторая строка (длина 44 символа):

Позиции 1–9: номер паспорта (буквы и цифры)

Позиция 10: контрольная цифра номера паспорта

Позиции 11–13: код гражданства (например, RUS)

Позиции 14–19: дата рождения (YYMMDD)

Позиция 20: контрольная цифра даты рождения

Позиция 21: пол (M/F/<)

Позиции 22–27: срок действия паспорта (YYMMDD)

Позиция 28: контрольная цифра срока действия

Позиции 29–42: личный номер (может быть заполнен <)

Позиция 43: контрольная цифра личного номера

Позиция 44: общая контрольная цифра (по всем полям второй строки)

Реализация парсера
python
import re
from datetime import datetime

class MRZParser:
    """Парсер MRZ по стандарту ICAO 9303 (TD3) для российских паспортов."""

    def __init__(self, line1: str, line2: str):
        self.line1 = line1
        self.line2 = line2
        self.raw_data = {}
        self.errors = []

    @staticmethod
    def clean_mrz_line(line: str) -> str:
        """Удаляет недопустимые символы, оставляет буквы, цифры, < и пробелы."""
        return re.sub(r'[^A-Z0-9<]', '', line.upper())

    def validate_line_length(self) -> bool:
        """Проверяет, что обе строки имеют длину 44 символа."""
        l1 = len(self.clean_mrz_line(self.line1))
        l2 = len(self.clean_mrz_line(self.line2))
        if l1 != 44 or l2 != 44:
            self.errors.append(f"Неверная длина строк: line1={l1}, line2={l2} (ожидается 44)")
            return False
        return True

    @staticmethod
    def compute_check_digit(data: str) -> int:
        """
        Вычисляет контрольную цифру для строки по алгоритму ICAO.
        Веса: 7, 3, 1, 7, 3, 1, ...
        """
        weights = [7, 3, 1]
        total = 0
        for i, ch in enumerate(data):
            if ch.isdigit():
                val = int(ch)
            elif ch.isalpha():
                val = ord(ch) - 55  # A=10, B=11, ..., Z=35
            else:
                val = 0  # '<' или другой символ
            total += val * weights[i % 3]
        return total % 10

    def parse(self) -> dict:
        """Основной метод разбора. Возвращает словарь с данными."""
        if not self.validate_line_length():
            return {"error": self.errors}

        line1 = self.clean_mrz_line(self.line1)
        line2 = self.clean_mrz_line(self.line2)

        # ---------- Первая строка ----------
        doc_type = line1[0:2]            # 'P<'
        issuer = line1[2:5]              # 'RUS'
        # Остальное: фамилия и имя
        remainder = line1[5:]
        # Разделитель << между фамилией и именем
        if '<<' not in remainder:
            self.errors.append("Не найдено разделителя '<<' между фамилией и именем")
            return {"error": self.errors}

        surname_part, given_names_part = remainder.split('<<', 1)
        # Заменяем < на пробелы и убираем лишние заполнители в конце
        surname = surname_part.replace('<', ' ').strip()
        # Имя + отчество: могут быть разделены '<'
        given_names = given_names_part.replace('<', ' ').strip()
        # Разделяем имя и отчество (если есть два слова)
        name_parts = given_names.split()
        first_name = name_parts[0] if name_parts else ''
        middle_name = name_parts[1] if len(name_parts) > 1 else ''

        # ---------- Вторая строка ----------
        # Номер паспорта (9 символов)
        passport_number = line2[0:9]
        passport_number_check = line2[9:10]
        if passport_number_check != str(self.compute_check_digit(passport_number)):
            self.errors.append("Контрольная цифра номера паспорта не совпадает")

        nationality = line2[10:13]  # 'RUS'

        # Дата рождения (YYMMDD)
        birth_date_raw = line2[13:19]
        birth_date_check = line2[19:20]
        if birth_date_check != str(self.compute_check_digit(birth_date_raw)):
            self.errors.append("Контрольная цифра даты рождения не совпадает")
        # Преобразуем в читаемый формат (при возможности)
        try:
            birth_date_obj = datetime.strptime(birth_date_raw, '%y%m%d')
            birth_date = birth_date_obj.strftime('%d.%m.%Y')
        except ValueError:
            birth_date = f"20{birth_date_raw[0:2]}-{birth_date_raw[2:4]}-{birth_date_raw[4:6]}" if birth_date_raw != '<<<<<<' else None
            if birth_date_raw == '<<<<<<':
                birth_date = None

        sex = line2[20:21]  # M, F, <
        if sex == '<':
            sex = None

        # Срок действия (YYMMDD)
        expiry_raw = line2[21:27]
        expiry_check = line2[27:28]
        if expiry_check != str(self.compute_check_digit(expiry_raw)):
            self.errors.append("Контрольная цифра срока действия не совпадает")
        try:
            expiry_obj = datetime.strptime(expiry_raw, '%y%m%d')
            expiry_date = expiry_obj.strftime('%d.%m.%Y')
        except ValueError:
            expiry_date = None

        # Личный номер (может быть пустым или содержать <<<)
        personal_number_raw = line2[28:42]
        personal_number_check = line2[42:43]
        personal_number = personal_number_raw.replace('<', '').strip()
        if personal_number and personal_number_check != str(self.compute_check_digit(personal_number_raw)):
            self.errors.append("Контрольная цифра личного номера не совпадает")

        # Общая контрольная цифра (проверка всей второй строки без последней цифры)
        total_check = line2[43:44]
        full_data_for_total = line2[0:43]
        if total_check != str(self.compute_check_digit(full_data_for_total)):
            self.errors.append("Общая контрольная цифра не совпадает")

        # Результат
        self.raw_data = {
            "doc_type": doc_type,
            "issuer": issuer,
            "surname": surname,
            "first_name": first_name,
            "middle_name": middle_name,
            "full_name": f"{surname} {first_name} {middle_name}".strip(),
            "passport_number": passport_number,
            "nationality": nationality,
            "birth_date": birth_date,
            "birth_date_raw": birth_date_raw,
            "sex": sex,
            "expiry_date": expiry_date,
            "personal_number": personal_number,
            "errors": self.errors
        }
        return self.raw_data

    def pretty_print(self):
        """Выводит данные в удобном виде."""
        data = self.parse()
        print("=== Результат разбора MRZ ===")
        print(f"Фамилия:          {data.get('surname')}")
        print(f"Имя:              {data.get('first_name')}")
        print(f"Отчество:         {data.get('middle_name')}")
        print(f"Номер паспорта:   {data.get('passport_number')}")
        print(f"Гражданство:      {data.get('nationality')}")
        print(f"Дата рождения:    {data.get('birth_date')}")
        print(f"Пол:              {data.get('sex')}")
        print(f"Срок действия:    {data.get('expiry_date')}")
        print(f"Личный номер:     {data.get('personal_number')}")
        if data.get('errors'):
            print(f"Ошибки:           {', '.join(data['errors'])}")
        else:
            print("Контрольные суммы: OK")


# Пример использования
if __name__ == "__main__":
    # Пример из вашего файла (Ягафаров)
    mrz_line1 = "PNRUS8GAFAROV<<RAMIS<RASIMOVI3<<<<<<<<<<<<<<<<"
    mrz_line2 = "8020980852RUS0506247M<<<<<<<<5250904020033<74"

    parser = MRZParser(mrz_line1, mrz_line2)
    parser.pretty_print()
Как это работает
Очистка строк – удаляются лишние символы, всё приводится к верхнему регистру.

Проверка длины – строгая проверка на 44 символа.

Вычисление контрольной цифры – по алгоритму с весами 7, 3, 1, повторяющимися циклически.

Разбор первой строки – выделяется тип документа, код эмитента, фамилия и имя (с возможным отчеством).

Разбор второй строки – извлекается номер паспорта, дата рождения, пол, срок действия, личный номер.

Проверка контрольных сумм – для номера паспорта, даты рождения, срока действия, личного номера и общей суммы. При несовпадении записывается ошибка, но данные всё равно возвращаются.

Вывод результата – в удобочитаемом виде.

Адаптация под российские внутренние паспорта
Для внутреннего паспорта РФ (формат TD1 – ID-карта) структура MRZ иная, но принцип похож. Если нужно, я могу предоставить и парсер для TD1 (3 строки по 30 символов). Но в ваших файлах использовался именно TD3 (паспорт), поэтому данный код полностью покрывает задачу.

Интеграция в ваш процесс
Вы можете:

Вставить этот парсер в скрипт, который читает PDF и извлекает MRZ-строки из нижней части страницы.

Использовать регулярное выражение для поиска двух строк по 44 символа, содержащих P<RUS и цифры.
